**4탄. Making a bar chart**
===================

1탄_drawing divs와 이어지는 단원이다.
이전보다 더 다양해진 표현 양식을 html로 보여주는 단계이다.

# 01_making a bar chart_divs.html


```
<!DOCTYPE html>
<html>

	<head>
		<meta charset = "utf-8">
		<title>D3: Making a bar chart </title>
		<script src="http://d3js.org/d3.v3.min.js"></script>
		<style type="text/css">

			div.bar {
				display : inline-block;
				width : 20px;
				height : 75px;
				margin-right : 2px;
				background-color: teal; //teal 청록색 같은 색으로 지정
			}

		</style>	


	</head>

	<body>

		<script type = "text/javascript">

			var dataset = [5, 10, 13, 19, 21, 25]

			d3.select("body").selectAll("div")
					.data(dataset) //요소에 데이터 바인딩
					.enter()
					.append("div") //div 요소를 추가
					.attr("class","bar") //문서요소의 속성을 설정할 때 사용
					.style("height", function(d) {
						var barHeight = d * 5;
						return barHeight + "px";
					});
		</script>
	</body>
</html>	
```


#2 making_a bar chart_rects.html

```
<!DOCTYPE html>
<html>

	<head>
		<meta charset = "utf-8">
		<title>D3: Making a bar chart </title>
		<script src="http://d3js.org/d3.v3.min.js"></script>
		<style type="text/css">


		</style>	


	</head>

	<body>

		<script type = "text/javascript">

			//Width and height
			var w = 500;
			var h = 100;

			var dataset = [ 5, 10, 13, 19, 21, 25, 22, 18, 15, 13,
							11, 12, 15, 20, 18, 17, 16, 18, 23, 25 ];


			//Create SVG element
			var svg = d3.select("body")
									.append("svg")
									.attr("width", w)
									.attr("height", h);

			svg.selectAll("rect")
				.data(dataset) //요소에 데이터 바인딩
				.enter()       //할당되지 않은 데이터를 자동으로 할당
				.append("rect") //rect 요소를 추가
				.attr("x",0)
				.attr("y",0)
				.attr("width",20)
				.attr("height",100);

			
		</script>
	</body>
</html>
```

#6 making_a bar chart_height.html
```
<!DOCTYPE html>
<html>

	<head>
		<meta charset = "utf-8">
		<title>D3: Making a bar chart </title>
		<script src="http://d3js.org/d3.v3.min.js"></script>

		</style>	


	</head>

	<body>

		<script type = "text/javascript">

			//Width and height
			var w = 500;
			var h = 100;
			var barPadding = 1;

			var dataset = [ 5, 10, 13, 19, 21, 25, 22, 18, 15, 13,
							11, 12, 15, 20, 18, 17, 16, 18, 23, 25 ];


			//Create SVG element
			var svg = d3.select("body")
									.append("svg")
```
